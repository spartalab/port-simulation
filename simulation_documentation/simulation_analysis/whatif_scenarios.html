 14def reduce_cranes(env, change, terminals, time_start, time_end, port_berths_container_terminals, terminal_data_cache, berths_aff_per):
 15    """
 16    This function reduces the number of cranes in each berth by `change` amount by removing cranes from the berth's crane store between `time_start` and `time_end`.
 17    It also restores the cranes after the reduction period.
 18    Args:
 19        env (simpy.Environment): The simulation environment.
 20        change (int): The number of cranes to be removed (negative value) or added (positive value).
 21        terminals (list): List of terminal indices where the cranes will be reduced.
 22        time_start (int): The time at which the reduction starts.
 23        time_end (int): The time at which the reduction ends.
 24        port_berths_container_terminals (dict): Dictionary containing berth information for container terminals.
 25        terminal_data_cache (dict): Cache containing terminal data for crane transfer rates.
 26        berths_aff_per (float): Percentage of berths to be affected by the change.
 27    Returns:
 28        None
 29    """
 30    port_berth_initially = {}
 31    for terminal_idx in terminals:
 32        port_berth_initially[terminal_idx] = []
 33        for berths in port_berths_container_terminals[terminal_idx].items:
 34            port_berth_initially[terminal_idx].append(berths)
 35
 36    print(
 37        f"Modelling disruption: Attempting to remove {-change} cranes from each berth in terminals {terminals} from time {time_start} to {time_end}")
 38
 39    yield env.timeout(time_start)
 40
 41    for terminal_idx in terminals:
 42        num_berths = len(port_berth_initially[terminal_idx])
 43        berths_aff = int(berths_aff_per*num_berths)
 44        count = 0
 45        for berth in port_berth_initially[terminal_idx]:
 46            count += 1
 47            if change < 0:
 48                num_to_remove = min(len(berth.cranes.items), abs(change))
 49                for _ in range(num_to_remove):
 50                    if berth.cranes.items:
 51                        crane = yield berth.cranes.get()
 52            if count >= berths_aff:
 53                break
 54
 55    yield env.timeout(time_end - time_start)
 56
 57    for terminal_idx in terminals:
 58        num_berths = len(port_berth_initially[terminal_idx])
 59        berths_aff = int(berths_aff_per*num_berths)
 60        count = 0
 61        for berth in port_berth_initially[terminal_idx]:
 62            count += 1
 63            if change < 0:
 64                for i in range(abs(change)):
 65                    restored_crane = Crane(id=f"restored_{berth.id}_{i}", width="width", crane_transfer_rate=get_value_by_terminal(
 66                        terminal_data_cache, 'Container', terminal_idx+1, 'transfer rate per unit'))
 67                    yield berth.cranes.put(restored_crane)
 68            if count >= berths_aff:
 69                break

193def reduce_berths(env, change, terminals, time_start, time_end, port_berths_terminal):
194    """
195    This function reduces the number of berths in each terminal by `change` amount by removing berths from the terminal's berth store between `time_start` and `time_end`.
196    It also restores the berths after the reduction period.
197    Args:
198        env (simpy.Environment): The simulation environment.
199        change (int): The number of berths to be removed (negative value) or added (positive value).
200        terminals (list): List of terminal indices where the berths will be reduced.        
201        time_start (int): The time at which the reduction starts.
202        time_end (int): The time at which the reduction ends.
203        port_berths_terminal (dict): Dictionary containing berth information for the terminals.
204    Returns:
205        None
206    """
207    print(
208        f"Modelling disruption: Attempting to remove {-change} berths from terminals {terminals} from time {time_start} to {time_end}")
209
210    yield env.timeout(time_start)
211
212    removed_berths = {terminal: [] for terminal in terminals}
213
214    # Reduce capacity by taking items out of the store
215    for terminal in terminals:
216        for _ in range(abs(change)):
217            # Remove a berth to simulate reduced capacity
218            berth = yield port_berths_terminal[terminal].get()
219            removed_berths[terminal].append(berth)
220        print(
221            f"Terminal {terminal}: Berth capacity effectively reduced by {abs(change)} at time {env.now}")
222
223    yield env.timeout(time_end - time_start)
224
225    # Restore the removed berths to the store
226    for terminal in terminals:
227        for berth in removed_berths[terminal]:
228            yield port_berths_terminal[terminal].put(berth)
229        print(
230            f"Terminal {terminal}: Berth Capacity restored by {abs(change)} at time {env.now}")

339def reduce_bay_capacity(env, reduction, terminals, start_time, end_time, port_loading_bays_liquid_terminals):
340    """
341    This function reduces the bay capacity in each terminal by `reduction` amount by holding requests for bays in the terminal's bay store between `start_time` and `end_time`.
342    It also restores the bay capacity after the reduction period.
343    Args:
344        env (simpy.Environment): The simulation environment.
345        reduction (int): The number of bays to be removed (negative value) or added (positive value).
346        terminals (list): List of terminal indices where the bay capacity will be reduced.
347        start_time (int): The time at which the reduction starts.
348        end_time (int): The time at which the reduction ends.
349        port_loading_bays_liquid_terminals (dict): Dictionary containing bay information for the terminals.
350    Returns:
351        None
352    """
353
354    print(
355        f"Modelling disruption: Attempting to remove {-reduction} bays from terminals {terminals} from time {start_time} to {end_time}")
356
357    yield env.timeout(start_time)
358
359    # Request `reduction` number of bays to reduce the effective capacity
360    requests = []
361    for terminal in terminals:
362        for _ in range(reduction):
363            request = port_loading_bays_liquid_terminals[terminal].request()
364            yield request  # Hold this request to reduce availability
365            requests.append((terminal, request))
366        print(
367            f"{reduction} units of bays removed from terminal {terminal} at {start_time}")
368    yield env.timeout(end_time - start_time)
369
370    # Release the requests to restore full capacity
371    for terminal, request in requests:
372        port_loading_bays_liquid_terminals[terminal].release(request)
373        print(
374            f"{reduction} units of bays restored from terminal {terminal} at {end_time}")

449def adjust_arrival_rate(start_time, end_time, rate_parameter, run_id, plot_input=True):
450    """
451    This function adjusts the arrival times of ships in the ship data CSV file by a given rate parameter.
452    It modifies the arrival times of ships that fall within the specified time range and saves the updated data back to the CSV file.
453    Args:
454        start_time (float): The start time of the range to adjust arrival times.
455        end_time (float): The end time of the range to adjust arrival times.
456        rate_parameter (float): The factor by which to adjust the arrival times.
457        run_id (str): The unique identifier for the simulation run.
458        plot_input (bool): Whether to plot the input data before and after adjustment.
459    returns:
460        ship_data_df (pd.DataFrame): The updated ship data DataFrame with adjusted arrival times.
461        ship_data (dict): The updated ship data as a dictionary.
462    """
463
464    ship_data = pd.read_csv(f".{run_id}/logs/ship_data.csv")
465    in_time_range = (ship_data['arrival'] >= start_time) & (
466        ship_data['arrival'] <= end_time)
467    ship_data['old_arrival'] = ship_data['arrival']
468    ship_data.loc[in_time_range, 'arrival'] /= rate_parameter
469
470    ship_data_df = ship_data.drop(columns=['interarrival'])
471    ship_data_df = ship_data_df.sort_values('arrival')
472    output_path = f'.{run_id}/logs/ship_data.csv'
473    ship_data_df.to_csv(output_path, index=False)
474    ship_data = ship_data_df.to_dict(orient="index")
475
476    if plot_input:
477        fig, axes = plt.subplots(1, 2, figsize=(14, 6))
478        axes[0].hist(ship_data_df['old_arrival'],
479                     bins=30, color='blue', alpha=0.7)
480        axes[0].set_xlabel('Arrival Time')
481        axes[0].set_ylabel('Frequency')
482        axes[0].set_title('Old Arrival Times')
483        axes[1].hist(ship_data_df['arrival'], bins=30,
484                     color='green', alpha=0.7)
485        axes[1].set_xlabel('Arrival Time')
486        axes[1].set_ylabel('Frequency')
487        axes[1].set_title('New Arrival Times')
488        min_x = min(ship_data_df['old_arrival'].min(),
489                    ship_data_df['arrival'].min())
490        max_x = max(ship_data_df['old_arrival'].max(),
491                    ship_data_df['arrival'].max())
492        min_y = 0
493        max_y = max(axes[0].get_ylim()[1], axes[1].get_ylim()[1])
494
495        for ax in axes:
496            ax.set_xlim(min_x, max_x)
497            ax.set_ylim(min_y, max_y)
498
499        plt.tight_layout()
500        plt.savefig(f".{run_id}/plots/scenarios/factor_arrival_scnario_input")
501        plt.close()
502
503    return ship_data_df, ship_data

506def stop_and_bulk_arrival(start_time, end_time, run_id, plot_input=False):
507    """
508    This function modifies the ship arrival times in the ship data CSV file by stopping arrivals between `start_time` and `end_time`.
509    It also simulates a bulk arrival scenario after `end_time` by distributing the arrivals evenly over a recovery period.
510    Args:
511        start_time (float): The start time of the range to stop arrivals.
512        end_time (float): The end time of the range to stop arrivals.
513        run_id (str): The unique identifier for the simulation run.
514        plot_input (bool): Whether to plot the input data before and after adjustment.
515    returns:
516        ship_data_df (pd.DataFrame): The updated ship data DataFrame with adjusted arrival times.
517        ship_data_dict (dict): The updated ship data as a dictionary.
518    """
519
520    ship_data = pd.read_csv(f'.{run_id}/logs/ship_data.csv')
521    ship_data['old_arrival'] = ship_data['arrival']
522    in_time_range = (ship_data['arrival'] >= start_time) & (
523        ship_data['arrival'] <= end_time)
524
525    num_ships = in_time_range.sum()
526    recovery_period = 24*5  # 5 days
527    gap = recovery_period/num_ships
528    print(
529        f"Modelling disruption: Stopping arrivals between {start_time} and {end_time} and bulk arrivals after {end_time} with a gap of {gap} hours")
